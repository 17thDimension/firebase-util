Getting Started

On the Web

<script src="https://cdn.firebase.com/js/client/2.1.1/firebase.js"></script>
<script src="https://cdn.firebase.com/libs/firebase-util/0.2.0/firebase-util.min.js"></script>
<script>
    Firebase.util.log('hello world');
</script>

Node.js

var Firebase = require('firebase');
var FirebaseUtil = require('./firebase-util.js');
FirebaseUtil.log('hello world');
